import java.util.Scanner;

public class StudentCRUD {

    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);

        try {
            Class.forName("com.mysql.cj.jdbc.Driver");

            Connection con = DriverManager.getConnection(
                    "jdbc:mysql://localhost:3306/college",
                    "root",

            );

            Statement stmt = con.createStatement();

            int choice;

            do {
                System.out.println("\n---- MENU ----");
                System.out.println("1. Insert");

                System.out.println("3. Display All");
                System.out.println("4. Update");
                System.out.println("5. Delete");
                System.out.println("6. Exit");

                choice = sc.nextInt();

                switch(choice) {

                    case 1:
                        System.out.println("Enter id, name, course, marks:");
                        int id = sc.nextInt();
                        String name = sc.next();
                        String course = sc.next();
                        int marks = sc.nextInt();

                        stmt.executeUpdate(
                            "INSERT INTO student VALUES(" +
                            id + ",'" + name + "','" +
                            course + "'," + marks + ")"
                        );
                        System.out.println("Record Inserted");
                        break;

                    case 2:
                        System.out.println("Enter id:");
                        int sid = sc.nextInt();

                        ResultSet rs = stmt.executeQuery(
                            "SELECT * FROM student WHERE id=" + sid
                        );

                        if(rs.next()) {
                            System.out.println(
                                rs.getInt("id") + " " +
                                rs.getString("name") + " " +
                                rs.getString("course") + " " +
                                rs.getInt("marks")
                            );
                        } else {
                            System.out.println("Record Not Found");
                        }
                        break;

                    case 3:
                        ResultSet rs2 = stmt.executeQuery(
                            "SELECT * FROM student"
                        );

                        while(rs2.next()) {
                            System.out.println(
                                rs2.getInt("id") + " " +
                                rs2.getString("name") + " " +
                                rs2.getString("course") + " " +
                                rs2.getInt("marks")
                            );
                        }
                        break;

                    case 4:
                        System.out.println("Enter id and new marks:");
                        int uid = sc.nextInt();
                        int newMarks = sc.nextInt();

                        int rows = stmt.executeUpdate(
                            "UPDATE student SET marks=" +
                            newMarks + " WHERE id=" + uid
                        );

                        System.out.println(rows + " record updated");
                        break;

                    case 5:
                        System.out.println("Enter id:");
                        int did = sc.nextInt();

                        int del = stmt.executeUpdate(
                            "DELETE FROM student WHERE id=" + did
                        );

                        System.out.println(del + " record deleted");
                        break;

                    case 6:
                        System.out.println("Exiting...");
                        break;

                    default:
                        System.out.println("Invalid Choice");
                }

            } while(choice != 6);

            con.close();
            stmt.close();
            sc.close();

        } catch(Exception e) {
            System.out.println(e);
        }
    }

import javax.swing.*;
import java.awt.*;
import java.sql.*;

public class Assignment3_3 extends JFrame {
    JTextField txtId, txtName, txtCourse, txtMarks;
    JButton btnFirst, btnPrev, btnNext, btnLast;
    ResultSet rs;

    public Assignment3_3() {
        setTitle("Student Navigation");
        setLayout(new GridLayout(5, 2, 10, 10));

        add(new JLabel("ID:")); txtId = new JTextField(); add(txtId);
        add(new JLabel("Name:")); txtName = new JTextField(); add(txtName);
        add(new JLabel("Course:")); txtCourse = new JTextField(); add(txtCourse);
        add(new JLabel("Marks:")); txtMarks = new JTextField(); add(txtMarks);

        btnFirst = new JButton("First"); btnPrev = new JButton("Previous");
        btnNext = new JButton("Next"); btnLast = new JButton("Last");
        add(btnFirst); add(btnPrev); add(btnNext); add(btnLast);

        initDB();

        btnFirst.addActionListener(e -> { try { rs.first(); showData(); } catch(Exception ex){} });
        btnLast.addActionListener(e -> { try { rs.last(); showData(); } catch(Exception ex){} });
        btnNext.addActionListener(e -> { try { if(rs.next()) showData(); else rs.last(); } catch(Exception ex){} });
        btnPrev.addActionListener(e -> { try { if(rs.previous()) showData(); else rs.first(); } catch(Exception ex){} });

        setSize(400, 300);
        setDefaultCloseOperation(EXIT_ON_CLOSE);
        setVisible(true);
    }

    private void initDB() {
        try {
            Class.forName("com.mysql.cj.jdbc.Driver");
            Connection con = DriverManager.getConnection("jdbc:mysql://localhost:3306/exp", "root", "root");
            Statement stmt = con.createStatement(ResultSet.TYPE_SCROLL_INSENSITIVE, ResultSet.CONCUR_READ_ONLY);
            rs = stmt.executeQuery("SELECT * FROM student");
            rs.next(); showData();
        } catch (Exception e) { e.printStackTrace(); }
    }

    private void showData() throws SQLException {
        txtId.setText(rs.getString("id"));
        txtName.setText(rs.getString("name"));
        txtCourse.setText(rs.getString("course"));
        txtMarks.setText(rs.getString("marks"));
    }

    public static void main(String[] args) { new Assignment3_3(); }
}

public class DiscountServlet extends HttpServlet {
	private static final long serialVersionUID = 1L;

	protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
		 HttpSession session = request.getSession();

	        String services[] = (String[]) session.getAttribute("services");

	        String discountStr = request.getParameter("discount");

	        session.setAttribute("discount", discountStr);

	        int total = 0;

	        PrintWriter out = response.getWriter();

	        out.println("<html><body>");

	        out.println("<h2>Selected Services</h2>");

	        if(services != null)
	        {
	            for(String service : services)
	            {
	                if(service.equals("Face Spa"))
	                {
	                    out.println("Face Spa : ₹500<br>");
	                    total += 500;
	                }

	                if(service.equals("Body Spa"))
	                {
	                    out.println("Body Spa : ₹1000<br>");
	                    total += 1000;
	                }

	                if(service.equals("Foot Spa"))
	                {
	                    out.println("Foot Spa : ₹300<br>");
	                    total += 300;
	                }
	            }
	        }

	        int discount = Integer.parseInt(discountStr);

	        double discountAmount = total * discount / 100.0;

	        double netAmount = total - discountAmount;

	        out.println("<br>Total = ₹" + total);

	        out.println("<br>Discount = " + discount + "%");

	        out.println("<br>Net Payable = ₹" + netAmount);

package com.shriyash.mca.service;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import com.shriyash.mca.model.Account;
import com.shriyash.mca.repository.AccountRepository;

@Service
public class AccountService {

    @Autowired
    private AccountRepository accountRepository;

    public Account createAccount(Account account) {
        return accountRepository.save(account);
    }

    public Account getAccount(Long id) {
        return accountRepository.findById(id).orElseThrow(() -> new RuntimeException("Account not found"));
    }

    @Transactional
    public void transferMoney(Long fromAccountId, Long toAccountId, Double amount, boolean simulateError) {
        Account fromAccount = accountRepository.findById(fromAccountId)
                .orElseThrow(() -> new RuntimeException("Source account not found"));
        
        Account toAccount = accountRepository.findById(toAccountId)
                .orElseThrow(() -> new RuntimeException("Destination account not found"));

        if (fromAccount.getBalance() < amount) {
            throw new RuntimeException("Insufficient balance");
        }

        fromAccount.setBalance(fromAccount.getBalance() - amount);
        accountRepository.save(fromAccount);

        if (simulateError) {
            throw new RuntimeException("Simulated error during money transfer! Transaction should rollback.");
        }

        toAccount.setBalance(toAccount.getBalance() + amount);
        accountRepository.save(toAccount);
    }
}

import org.springframework.web.bind.annotation.ControllerAdvice;
import org.springframework.web.bind.annotation.ExceptionHandler;

@ControllerAdvice
public class GlobalExceptionHandler {

    @ExceptionHandler(StudentNotFoundException.class)
    public ResponseEntity<Object> handleStudentNotFoundException(StudentNotFoundException ex) {
        Map<String, Object> body = new HashMap<>();
        body.put("timestamp", LocalDateTime.now());
        body.put("status", HttpStatus.NOT_FOUND.value());
        body.put("error", "Not Found");
        body.put("message", ex.getMessage());

        return new ResponseEntity<>(body, HttpStatus.NOT_FOUND);
    }

import React, { useEffect, useState } from "react";
import "./App.css";

function App() {
  const [students, setStudents] = useState([]);
  const [id, setId] = useState("");
  const [name, setName] = useState("");
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    fetch("http://localhost:8080/McaStudents")
      .then((res) => res.json())
      .then((data) => {
        setStudents(data);
        setLoading(false);
      })
      .catch((err) => {
        console.error("Error fetching students:", err);
        setLoading(false);
      });
  }, []);

  const addStudent = () => {
    if (!id || !name) {
      alert("Please enter both ID and Name");
      return;
    }

    fetch("http://localhost:8080/McaStudents", {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
      },
      body: JSON.stringify({
        id: Number(id),
        name: name,
      }),
    })
      .then((res) => res.text())
      .then((msg) => {
        alert(msg);
        window.location.reload();
      })
      .catch((err) => {
        console.error("Error adding student:", err);
        alert("Failed to add student");
      });
  };

  return (
    <div className="container">
      <div className="glass-card">
        <header>
          <h1>MCA Students</h1>
          <p className="subtitle">Management System</p>
        </header>

        <div className="form-section">
          <div className="input-group">
            <input
              type="number"
              placeholder="Student ID"
              value={id}
              onChange={(e) => setId(e.target.value)}
              id="student-id"
            />
            <label htmlFor="student-id">ID</label>
          </div>

          <div className="input-group">
            <input
              type="text"
              placeholder="Student Name"
              value={name}
              onChange={(e) => setName(e.target.value)}
              id="student-name"
            />
            <label htmlFor="student-name">Name</label>
          </div>

          <button className="add-btn" onClick={addStudent}>
            <span>Add Student</span>
          </button>
        </div>

        <hr className="divider" />

        <div className="list-section">
          <h2>Student Directory</h2>
          {loading ? (
            <div className="loader">Loading...</div>
          ) : students.length > 0 ? (
            <ul className="student-list">
              {students.map((s) => (
                <li key={s.id} className="student-item">
                  <span className="sid">#{s.id}</span>
                  <span className="sname">{s.name}</span>
                </li>
              ))}
            </ul>
          ) : (
            <p className="empty-msg">No students found.</p>
          )}
        </div>
      </div>
    </div>
  );
}

export default App;

import { useState } from "react";
import axios from "axios";

function App() {
  const [accounts, setAccounts] = useState([]);
  const [showAccounts, setShowAccounts] = useState(false);
  const [name, setName] = useState("");
  const [balance, setBalance] = useState("");

  // View Accounts
  const loadAccounts = async () => {
    try {
      const response = await axios.get("http://localhost:8082/bank/show");
      setAccounts(response.data);
    } catch (error) {
      console.log(error);
    }
  };

  // Create Account
  const createAccount = async (e) => {
    e.preventDefault();
    try {
      await axios.post("http://localhost:8082/bank/create", {
        name: name,
        balance: balance
      });
      alert("Account Created Successfully");
      setName("");
      setBalance("");
    } catch (error) {
      console.log(error);
    }
  };

  return (
    <div style={{ padding: "20px" }}>
      <h1>Bank Management System</h1>
      {/* Create Account Form */}
      <h2>Create Account</h2>
      <form onSubmit={createAccount}>
        <input
          type="text"
          placeholder="Enter Name"
          value={name}
          onChange={(e) => setName(e.target.value)}
        />
        <br /><br />
        <input
          type="number"
          placeholder="Enter Balance"
          value={balance}
          onChange={(e) => setBalance(e.target.value)}
        />
        <br /><br />
        <button type="submit">Create Account</button>
      </form>
      <hr />
      {/* View Accounts */}
      <button
        onClick={() => {
          loadAccounts();
          setShowAccounts(true);
        }}
      >
        View All Accounts
      </button>
      <button
        onClick={() => setShowAccounts(false)}
        style={{ marginLeft: "10px" }}
      >
        Minimize Accounts
      </button>
      <br /><br />
      {/* Display Accounts */}
      {showAccounts &&
        accounts.map((acc) => (
          <div
            key={acc.id}
            style={{
              border: "1px solid black",
              padding: "10px",
              marginBottom: "10px"
            }}
          >
            <h3>{acc.name}</h3>
            <p>ID: {acc.id}</p>
            <p>Balance: {acc.balance}</p>
          </div>
        ))}
    </div>
  );
}

export default App;
